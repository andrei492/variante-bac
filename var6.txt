#include <iostream>
using namespace std;

int main() {
    int n, s = -1;
    cin >> n;
    while (n > 0) {
        if (n % 10 > s) {
            s = n % 10;
        }
        n = n / 10;
    }
    cout << s;
    return 0;
}
