#include <iostream>
using namespace std;
int main(){
    int n, m, mat[10][10];
    cin >> n >> m;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            if (i > j)
            {
                mat[i][j] = i+1;
            }
            else
            {
                mat[i][j] = j+1;
            }
        }
    }
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            cout << mat[i][j] << " ";
        }
        cout << endl;
    }
}

#include <iostream>
using namespace std;

int f(int a){
    for (int i = 2; i*i <= a; i++)
    {
        if (a % i == 0)
        {
            return i;
        }
        
    }
    return a;
}
int main(){
    int v[100], n, x, contor = 0;
    cin >> n;
    while (n)
    {
        cin  >> x;
        if (x == f(x))  
        {
            v[contor] =  x;
            contor++;
        }
        n--;
    }
    for (int i = 0; i < contor; i++)
    {
        for (int j = 0; j < contor; j++)
        {
            if (v[i] < v[j])
            {
                swap(v[i], v[j]);
            }
            
        }
        
    }
    for (int i = 0; i < contor; i++)
    {
        cout << v[i] << " ";
    }
}
