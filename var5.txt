#include <iostream>
using namespace std;
int main(){
    int n, m, mat[10][10];
    cin >> n >> m;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            if (i > j)
            {
                mat[i][j] = i+1;
            }
            else
            {
                mat[i][j] = j+1;
            }
        }
    }
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            cout << mat[i][j] << " ";
        }
        cout << endl;
    }
}
